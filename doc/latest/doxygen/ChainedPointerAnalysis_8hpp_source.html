266 static size_t stride(DYLD_CHAINED_PTR_FORMAT fmt) {
267 switch (fmt) {
268 case DYLD_CHAINED_PTR_FORMAT::NONE:
269 return 0;
270 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E:
271 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_USERLAND:
272 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_USERLAND24:
273 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_SHARED_CACHE:
274 return 8;
275
276 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_KERNEL:
277 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_FIRMWARE:
278 case DYLD_CHAINED_PTR_FORMAT::PTR_32_FIRMWARE:
279 case DYLD_CHAINED_PTR_FORMAT::PTR_64:
280 case DYLD_CHAINED_PTR_FORMAT::PTR_64_OFFSET:
281 case DYLD_CHAINED_PTR_FORMAT::PTR_32:
282 case DYLD_CHAINED_PTR_FORMAT::PTR_32_CACHE:
283 case DYLD_CHAINED_PTR_FORMAT::PTR_64_KERNEL_CACHE:
284 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_SEGMENTED:
285 return 4;
286
287 case DYLD_CHAINED_PTR_FORMAT::PTR_X86_64_KERNEL_CACHE:
288 return 1;
289 }
290 return 0;
291 }
292
293 static size_t ptr_size(DYLD_CHAINED_PTR_FORMAT fmt) {
294 switch (fmt) {
295 case DYLD_CHAINED_PTR_FORMAT::NONE:
296 return 0;
297 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E:
298 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_USERLAND:
299 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_USERLAND24:
300 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_KERNEL:
301 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_FIRMWARE:
302 case DYLD_CHAINED_PTR_FORMAT::PTR_64:
303 case DYLD_CHAINED_PTR_FORMAT::PTR_64_OFFSET:
304 case DYLD_CHAINED_PTR_FORMAT::PTR_64_KERNEL_CACHE:
305 case DYLD_CHAINED_PTR_FORMAT::PTR_X86_64_KERNEL_CACHE:
306 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_SHARED_CACHE:
307 case DYLD_CHAINED_PTR_FORMAT::PTR_ARM64E_SEGMENTED:
308 return sizeof(uint64_t);
309
310 case DYLD_CHAINED_PTR_FORMAT::PTR_32_FIRMWARE:
311 case DYLD_CHAINED_PTR_FORMAT::PTR_32:
312 case DYLD_CHAINED_PTR_FORMAT::PTR_32_CACHE:
313 return sizeof(uint32_t);
314 }
315 return 0;
316 }
