233 template<class T>
234 LIEF::result<T> get_int_from_virtual_address(
235 uint64_t va, VA_TYPES addr_type = VA_TYPES::AUTO) const
236 {
237 T value;
238 static_assert(std::is_integral<T>::value, "Require an integral type");
239 span<const uint8_t> raw = get_content_from_virtual_address(va, sizeof(T), addr_type);
240 if (raw.empty() || raw.size() < sizeof(T)) {
241 return make_error_code(lief_errors::read_error);
242 }
243
244 std::copy(raw.data(), raw.data() + sizeof(T),
245 reinterpret_cast<uint8_t*>(&value));
246 return value;
247 }
