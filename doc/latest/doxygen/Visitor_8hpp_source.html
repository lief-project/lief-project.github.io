633 template<class T>
634 void Visitor::dispatch(const T& obj) {
635  size_t hash = reinterpret_cast<size_t>(&obj);
636  if (this->visited_.find(hash) != std::end(this->visited_)) {
637  // Already visited
638  return;
639  }
640 
641  this->visited_.insert(hash);
642  this->visit(obj);
643 }
