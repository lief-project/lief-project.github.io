658template<typename Arg1, typename... Args>
659void Visitor::operator()(Arg1&& arg1, Args&&... args) {
660 dispatch(std::forward<Arg1>(arg1));
661 operator()(std::forward<Args>(args)...);
662}
663
664template<class T>
665void Visitor::dispatch(const T& obj) {
666 auto hash = reinterpret_cast<size_t>(&obj);
667 if (visited_.find(hash) != std::end(visited_)) {
668 // Already visited
669 return;
670 }
671
672 visited_.insert(hash);
673 visit(obj);
674}
