259 FILE_TYPE_DETAILS file_type_details() const {
260 if (file_subtype == 0) {
261 return FILE_TYPE_DETAILS::UNKNOWN;
262 }
263
264 auto ty = FILE_TYPE(file_type);
265 if (ty == FILE_TYPE::DRV) {
266 return FILE_TYPE_DETAILS(file_subtype | DRV_K);
267 }
268
269 if (ty == FILE_TYPE::FONT) {
270 return FILE_TYPE_DETAILS(file_subtype | FONT_K);
271 }
272
273 return FILE_TYPE_DETAILS::UNKNOWN;
274 }
